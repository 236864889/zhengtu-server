USE FLServerDB;

-- 补充缺少的 BALANCE、RECHARGEORDERS（若已存在可略过）
CREATE TABLE IF NOT EXISTS `BALANCE` (
  `ACCID`       INT UNSIGNED NOT NULL,
  `ACCOUNT`     VARCHAR(48)  NOT NULL,
  `ALLMONEYIN`  INT NOT NULL DEFAULT 0,
  `ALLMONEYOUT` INT NOT NULL DEFAULT 0,
  `MONEY`       INT NOT NULL DEFAULT 0,
  `ALLGOLDIN`   INT NOT NULL DEFAULT 0,
  `ALLGOLDOUT`  INT NOT NULL DEFAULT 0,
  `GOLD`        INT NOT NULL DEFAULT 0,
  `LASTTIME`    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`ACCID`),
  KEY `idx_account` (`ACCOUNT`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

手动部署 PHPMailer 源码（邮件验证码）

cd /www/wwwroot/hhemall

# 1) 安装 unzip（若没有）
yum install -y unzip || apt-get update && apt-get install -y unzip

# 2) 下载 PHPMailer 发行包（任选一个版本，示例 v6.9.1）
curl -L -o phpmailer.zip https://codeload.github.com/PHPMailer/PHPMailer/zip/refs/tags/v6.9.1

# 3) 解压并放到固定位置
unzip -q phpmailer.zip
rm -f phpmailer.zip
# 解压目录名类似 PHPMailer-6.9.1
mv PHPMailer-6.9.1 PHPMailer

# 重要：确保结构是这样（register_compat.php 同级）：
# /www/wwwroot/hhemall/PHPMailer/src/PHPMailer.php
# /www/wwwroot/hhemall/PHPMailer/src/SMTP.php
# /www/wwwroot/hhemall/PHPMailer/src/Exception.php
